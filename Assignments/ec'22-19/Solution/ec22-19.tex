\section{question}
Consider the $2$-bit multiplexer(MUX) shown in the figure. For OUTPUT to be the XOR of $C$ and $D$, the values of $A_0$,$A_1$,$A_2$ and $A_3$ are:
\begin{enumerate}
    \item [(A)] $A_0=0$, $A_1=0$, $A_2=1$, $A_3=1$
    \item [(B)] $A_0=1$, $A_1=0$, $A_2=1$, $A_3=0$
    \item [(C)] $A_0=0$, $A_1=1$, $A_2=1$, $A_3=0$
    \item [(D)] $A_0=1$, $A_1=1$, $A_2=0$, $A_3=0$
\end{enumerate}

\section{Solution}
Let Output be denoted by $Y$, then:
\begin{align}
    Y &= \overline{C}.\overline{D}.A_{0} + \overline{C}.D.A_{1} + C.\overline{D}.A_{2} + C.D.A_{3}
\end{align}
Given \begin{align}
    y &= C\oplus D
\end{align}
Then
\begin{align}
    A_0&=0\\
    A_1&=1\\
    A_2&=1\\
    A_3&=0
\end{align}
